<script>
var App = {
step: 'capture',
image: null,
points: [
{ id: 'head', label: '', x: 50, y: 15, color: '#ef4444' },
{ id: 'shoulder', label: '', x: 50, y: 30, color: '#f59e0b' },
{ id: 'back', label: '', x: 50, y: 50, color: '#eab308' },
{ id: 'waist', label: '', x: 50, y: 65, color: '#22c55e' },
{ id: 'hip', label: '', x: 50, y: 80, color: '#3b82f6' }
],
dragging: null,
result: null,
cameraMode: false,
stream: null,
render: function() {
this.renderStepIndicator();
this.renderContent();
},
renderStepIndicator: function() {
var steps = ['capture', 'analyze', 'result'];
var currentIndex = steps.indexOf(this.step);
var html = '';
for (var i = 0; i < steps.length; i++) {
var status = i < currentIndex ? 'completed' : i === currentIndex ? 'current' : 'upcoming';
html += '<div class="step ' + status + '">' + (status === 'completed' ? '' : (i + 1)) + '</div>';
if (i < steps.length - 1) {
html += '<div class="step-line ' + (i < currentIndex ? 'completed' : '') + '"></div>';
}
}
document.getElementById('stepIndicator').innerHTML = html;
},
renderContent: function() {
var content = document.getElementById('content');
if (this.step === 'capture' && !this.image) {
content.innerHTML = this.renderCaptureStep();
} else if (this.step === 'capture' && this.image) {
content.innerHTML = this.renderImagePreview();
} else if (this.step === 'analyze') {
content.innerHTML = this.renderAnalyzeStep();
var self = this;
setTimeout(function() { self.setupCanvas(); }, 100);
} else if (this.step === 'result') {
content.innerHTML = this.renderResultStep();
}
this.attachEventListeners();
},
renderCaptureStep: function() {
var self = this;
if (!this.cameraMode) {
return '<div class="info-box">' +
'<h3> </h3>' +
'<ul>' +
'<li> 20cm</li>' +
'<li> </li>' +
'<li> </li>' +
'<li> </li>' +
'</ul>' +
'</div>' +
'<button class="btn btn-primary" onclick="App.startCamera()"> </button>' +
'<button class="btn btn-secondary" onclick="document.getElementById(\'fileInput\').click()"> </button>' +
'<input type="file" id="fileInput" accept="image/*" class="hidden" onchange="App.handleFileUpload(event)">';
} else {
return '<video id="video" autoplay playsinline style="width: 100%; border-radius: 10px; margin-bottom: 15px;"></video>' +
'<button class="btn btn-primary" onclick="App.capturePhoto()"></button>' +
'<button class="btn btn-secondary" onclick="App.cancelCamera()"></button>';
}
},
renderImagePreview: function() {
return '<button class="btn btn-primary" onclick="App.step=\'analyze\'; App.render();"></button>' +
'<button class="btn btn-secondary" onclick="App.image=null; App.render();"></button>' +
'<img src="' + this.image + '" alt="Captured" class="preview-img">';
},
renderAnalyzeStep: function() {
return '<div class="info-box warning">' +
'<h3> </h3>' +
'<p>5</p>' +
'</div>' +
'<div id="canvas">' +
'<img id="canvasImg" src="' + this.image + '" alt="Analysis">' +
'<svg class="line-svg" id="linesSvg"></svg>' +
'<div id="markers"></div>' +
'</div>' +
'<button class="btn btn-success" onclick="App.analyzePosture()"></button>' +
'<button class="btn btn-secondary" onclick="App.step=\'capture\'; App.render();"></button>';
},
renderResultStep: function() {
if (!this.result) return '';
return '<div class="result-header">' +
'<h2></h2>' +
'<div class="body-type">' + this.result.bodyType + '</div>' +
'</div>' +
'<div class="description-box">' +
'<p>' + this.result.description + '</p>' +
'</div>' +
'<div class="result-grid">' +
'<div class="result-card">' +
'<h3> </h3>' +
'<div class="result-item"><span>:</span><span>' + this.result.pillowHeight + '</span></div>' +
'<div class="result-item"><span>:</span><span>' + this.result.pillowFirmness + '</span></div>' +
'</div>' +
'<div class="result-card mattress">' +
'<h3> </h3>' +
'<div class="result-item"><span>:</span><span>' + this.result.mattressFirmness + '</span></div>' +
'</div>' +
'</div>' +
'<div class="info-box">' +
'<h3> </h3>' +
'<p></p>' +
'</div>' +
'<button class="btn btn-primary" id="resetApp" onclick="App.reset()"> </button>';
},
attachEventListeners: function() {
// onclick
//
},
setupCanvas: function() {
this.renderLines();
this.renderMarkers();
this.attachDragListeners();
},
renderLines: function() {
var svg = document.getElementById('linesSvg');
if (!svg) return;
var lines = '';
for (var i = 0; i < this.points.length - 1; i++) {
var point = this.points[i];
var next = this.points[i + 1];
lines += '<line x1="' + point.x + '%" y1="' + point.y + '%" x2="' + next.x + '%" y2="' + next.y + '%" stroke="#6366f1" stroke-width="3" stroke-dasharray="5,5"/>';
}
svg.innerHTML = lines;
},
renderMarkers: function() {
var container = document.getElementById('markers');
if (!container) return;
var html = '';
for (var i = 0; i < this.points.length; i++) {
var p = this.points[i];
html += '<div class="point-marker" data-point="' + p.id + '" style="left: ' + p.x + '%; top: ' + p.y + '%;">' +
'<div class="point-circle" style="background-color: ' + p.color + ';"></div>' +
'<div class="point-label">' + p.label + '</div>' +
'</div>';
}
container.innerHTML = html;
},
attachDragListeners: function() {
var canvas = document.getElementById('canvas');
if (!canvas) return;
var markers = document.querySelectorAll('.point-marker');
var self = this;
for (var i = 0; i < markers.length; i++) {
var marker = markers[i];
var pointId = marker.getAttribute('data-point');
(function(pid) {
marker.addEventListener('mousedown', function(e) {
e.preventDefault();
self.dragging = pid;
});
marker.addEventListener('touchstart', function(e) {
e.preventDefault();
self.dragging = pid;
});
})(pointId);
}
var handleMove = function(e) {
if (!self.dragging) return;
e.preventDefault();
var rect = canvas.getBoundingClientRect();
var clientX, clientY;
if (e.type.indexOf('touch') !== -1) {
clientX = e.touches[0].clientX;
clientY = e.touches[0].clientY;
} else {
clientX = e.clientX;
clientY = e.clientY;
}
var x = ((clientX - rect.left) / rect.width) * 100;
var y = ((clientY - rect.top) / rect.height) * 100;
for (var i = 0; i < self.points.length; i++) {
if (self.points[i].id === self.dragging) {
self.points[i].x = Math.max(0, Math.min(100, x));
self.points[i].y = Math.max(0, Math.min(100, y));
}
}
self.renderLines();
self.renderMarkers();
self.attachDragListeners();
};
var stopDrag = function() {
self.dragging = null;
};
canvas.addEventListener('mousemove', handleMove);
canvas.addEventListener('touchmove', handleMove);
canvas.addEventListener('mouseup', stopDrag);
canvas.addEventListener('touchend', stopDrag);
canvas.addEventListener('mouseleave', stopDrag);
},
handleFileUpload: function(e) {
var file = e.target.files[0];
if (file) {
var reader = new FileReader();
var self = this;
reader.onload = function(event) {
self.image = event.target.result;
self.render();
};
reader.readAsDataURL(file);
}
},
startCamera: function() {
var self = this;
if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) {
alert('');
return;
}
navigator.mediaDevices.getUserMedia({
video: { facingMode: 'environment', width: 1280, height: 720 }
}).then(function(mediaStream) {
self.stream = mediaStream;
self.cameraMode = true;
self.render();
setTimeout(function() {
var video = document.getElementById('video');
if (video) {
video.srcObject = mediaStream;
}
}, 100);
}).catch(function(err) {
alert(': ' + err.message);
});
},
capturePhoto: function() {
var video = document.getElementById('video');
var canvas = document.createElement('canvas');
canvas.width = video.videoWidth;
canvas.height = video.videoHeight;
var ctx = canvas.getContext('2d');
ctx.drawImage(video, 0, 0);
this.image = canvas.toDataURL('image/jpeg');
this.stopCamera();
this.cameraMode = false;
this.render();
},
cancelCamera: function() {
this.stopCamera();
this.cameraMode = false;
this.render();
},
stopCamera: function() {
if (this.stream) {
var tracks = this.stream.getTracks();
for (var i = 0; i < tracks.length; i++) {
tracks[i].stop();
}
this.stream = null;
}
},
analyzePosture: function() {
var head = this.points[0];
var shoulder = this.points[1];
var back = this.points[2];
var waist = this.points[3];
var hip = this.points[4];
var shoulderCurve = Math.abs(shoulder.x - head.x);
var backCurve = Math.abs(back.x - shoulder.x);
var waistCurve = Math.abs(waist.x - back.x);
var hipCurve = Math.abs(hip.x - waist.x);
var totalCurve = shoulderCurve + backCurve + waistCurve + hipCurve;
var maxCurve = Math.max(shoulderCurve, backCurve, waistCurve, hipCurve);
var bodyType, pillowHeight, pillowFirmness, mattressFirmness, description;
if (totalCurve < 15) {
bodyType = 'S';
pillowHeight = '';
pillowFirmness = 'SOFT';
mattressFirmness = '';
description = 'S';
} else if (shoulderCurve === maxCurve || backCurve === maxCurve) {
bodyType = '';
pillowHeight = '';
pillowFirmness = 'REGULAR';
mattressFirmness = '';
description = '';
} else {
bodyType = '';
pillowHeight = '';
pillowFirmness = 'HARD';
mattressFirmness = '';
description = '';
}
this.result = {
bodyType: bodyType,
pillowHeight: pillowHeight,
pillowFirmness: pillowFirmness,
mattressFirmness: mattressFirmness,
description: description
};
this.step = 'result';
this.render();
},
reset: function() {
this.step = 'capture';
this.image = null;
this.result = null;
this.stopCamera();
this.cameraMode = false;
this.points = [
{ id: 'head', label: '', x: 50, y: 15, color: '#ef4444' },
{ id: 'shoulder', label: '', x: 50, y: 30, color: '#f59e0b' },
{ id: 'back', label: '', x: 50, y: 50, color: '#eab308' },
{ id: 'waist', label: '', x: 50, y: 65, color: '#22c55e' },
{ id: 'hip', label: '', x: 50, y: 80, color: '#3b82f6' }
];
this.render();
}
};
window.onload = function() {
App.render();
};
window.onbeforeunload = function() {
if (App.stream) {
App.stopCamera();
}
};
</script>
